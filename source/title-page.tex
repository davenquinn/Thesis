   \begin{center}
     {\SingleSpacing\Large\title\par}
     \titleskip
     Thesis by\\
     {\large \author}\par
     \authorskip
     In Partial Fulfillment of the Requirements for the\\
     Degree of\\
     \degreeaward\par
     \ifstrequal{#1}{logo}{

       \includegraphics[width=2in]{\@unilogo} \\
       \vspace{\stretch{1}}
     }{\vfill}
      \MakeUppercase{\university}\\
     \address\par
     \authorskip
     \copyyear\\
     {\normalsize Defended \defenddate\par}
   \end{center}
   \vspace*{.5in}
 \end{titlingpage}
